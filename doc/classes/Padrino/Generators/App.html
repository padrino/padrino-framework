      def create_app
        self.destination_root = options[:root]
        @app_name = name.gsub(/\W/, "_").underscore.camelize
        if in_app_root?
          self.behavior = :revoke if options[:destroy]
          app_skeleton(name, options[:tiny])
          empty_directory destination_root("public/#{name}")
          append_file destination_root("config/apps.rb"),  "\nPadrino.mount(\"#{@app_name}\").to(\"/#{name.underscore}\")"

          return if self.behavior == :revoke
          say ("\n=================================================================\nYour \#{@app_name} application has been installed.\nThis application has been mounted to /\#{name.underscore}\nYou can configure a different path by editing 'config/apps.rb'\n=================================================================\n\n").gsub(/ {10}/,'')
        else
          say "You are not at the root of a Padrino application! (config/boot.rb not found)"
        end
      end
